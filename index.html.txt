    <div id="reportsContainer">جاري تحميل البيانات...</div>

    <script>
        fetch('http://127.0.0.1:5001/reports')
        .then(res => res.json())
        .then(data => {
            let html = '<ul>';
            for (let key in data) {
                html += `<li>الموقع: ${data[key].Location} | الوصف: ${data[key].Description}</li>`;
            }
            html += '</ul>';
            document.getElementById('reportsContainer').innerHTML = html;
        });
    </script>
